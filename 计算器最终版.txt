    <!--将按键内容以字符串形式存储在文字框中当按钮为“=”时，调用eval方法计算结果然后将结果输出文字框中-->
    <script type="text/javascript">
        var numresult;
        var str;
        function onclicknum(nums) {
            str = document.getElementById("nummessege");
            str.value = str.value + nums;

        }
        function onclickclear() {
            str = document.getElementById("nummessege");
            str.value = "";
        }
        function onclickresult() {
            str = document.getElementById("nummessege");

            try{
                numresult = eval(str.value);  //执行该方法
            }
            catch(err)
            {
                numresult="INPUT ERROR";
            }


            str.value = numresult;
            if(str.value=="Infinity") str.value="VALUE ERROR";


        }
        function onclickBackspace() {
            str = document.getElementById("nummessege");
            num=str.value;
            var newstr=num.substring(0,num.length-1);
            str.value = newstr;
        }
    </script>
